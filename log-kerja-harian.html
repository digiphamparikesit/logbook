Pengaturan Sinkronisasi Google Sheets

Hubungkan aplikasi ini ke Google Sheets memakai Google Apps Script sebagai jembatan. Ikuti langkah ini sekali saja:

  1. Buka Google Sheets baru, buat sheet (boleh kosong).
  2. Menu Extensions → Apps Script.
  3. Hapus kode default, tempel kode di bawah ini.
  4. Klik Deploy → New deployment → Web app. Pilih "Execute as: Me" dan "Who has access: Anyone".
  5. Salin URL Web App yang muncul, tempel ke kolom di bawah, lalu klik Simpan.
function doPost(e) { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Data') || ss.insertSheet('Data'); if (sheet.getLastRow() === 0) { sheet.appendRow(['ID','Tanggal','Waktu','Nama Tugas','Kategori','Prioritas','Tenggat','Status','Durasi (menit)','Catatan','Diperbarui']); } var rows = sheet.getDataRange().getValues(); var rowIndex = -1; for (var i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { rowIndex = i + 1; break; } } if (data.action === 'delete') { if (rowIndex > -1) sheet.deleteRow(rowIndex); return ContentService.createTextOutput('deleted'); } var rowData = [data.id, data.date, data.time, data.name, data.category, data.priority, data.deadline, data.status, data.durationMinutes, data.note, new Date()]; if (rowIndex > -1) { sheet.getRange(rowIndex, 1, 1, rowData.length).setValues([rowData]); } else { sheet.appendRow(rowData); } return ContentService.createTextOutput('ok'); }
